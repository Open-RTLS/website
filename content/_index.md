  - acronym: "OMLOX"

    link: "https://omlox.com/"
    text: "OMLOX defines interoperable RTLS interfaces, including hub APIs and core-zone compatibility across vendors."
  - acronym: "OGC IMDF"
    name: "Open Geospatial Consortium Indoor Mapping Data Format"
    link: "https://www.ogc.org/standards/indoor-mapping-data-format/"
    text: "IMDF is an indoor mapping data model for consistent, standards-aligned venue maps."
  - acronym: "MQTT"
    name: "Message Queuing Telemetry Transport"
    link: "https://mqtt.org/mqtt-specification/"
    text: "MQTT is a lightweight publish/subscribe messaging standard widely used for event and telemetry pipelines."
components:
  - title: "OGC IMDF Editor"
    description: "An open editor for creating and managing [OGC IMDF](https://www.ogc.org/standards/indoor-mapping-data-format/) (Open Geospatial Consortium Indoor Mapping Data Format) indoor mapping data with practical workflows for operations teams and integrators."
  - title: "IMDF Validator"
    description: "A standards-aligned validator that helps teams detect and fix IMDF issues early in the delivery lifecycle."
  - title: "Open RTLS Hub (OMLOX API Compatible)"
    description: "A hub component exposing [OMLOX](https://omlox.com/) (open locating standard) API compatible interfaces for interoperable location data exchange."
  - title: "MQTT Support"
    description: "Built-in [MQTT](https://mqtt.org/mqtt-specification/) (Message Queuing Telemetry Transport) connectivity for ingesting and distributing events and telemetry in real time."
  - title: "Maplibre IMDF Map SDK"
    description: "Map capabilities aligned with OGC standards for geospatial and indoor location visualization."
  - title: "Connector Framework"
    description: "A modular connector framework for integrating RTLS hardware vendors, software systems, and enterprise workflows."
